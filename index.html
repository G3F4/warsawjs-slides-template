                            user {
                                id
                                name @skip(if: $noName)
                                loggedIn $include(if: $logged)
                            }
                        }

Type system

Type system

    SCALARS

                            
                            Int => 37
                            Float => 3.14
                            String => "Ninja!"
                            Boolean => true
                            ID => "d2#93^m8#d2023#23"

    ENUMS

                            
                            enum NUMBERS {
                              ONE
                              TWO
                              THREE
                            }

    NON-NULL

                            
                            !Int => 37 // coerce to non-empty value

    LISTS

                            
                            [String] => ["apple", "orange"]

    OBJECTS

                            
                            type UserType {
                                id: String!
                                name: String
                                loggedIn: Boolean
                            }

Type system

    INTERFACES

                            
                            interface User {
                                id: ID!
                            }
                            type Player implements User {
                                id: ID!
                                nick: String!
                            }
                            type Admin implements User {
                                id: ID!
                                secrets: [String!]
                            }

    UNIONS

                            
                            union Fruits = Apple | Orange | Plum

    INPUT OBJECTS

                            
                            input UserInput {
                              id: ID!
                              visible: Boolean
                            }

    INPUT OBJECTS USAGE EXAMPLE

                            
                            // JSON: { id: "1", visible: true }
                            mutation ($input: UserInput!) {
                              logUser(input: $input) {
                                id
                                loggedIn
                              }
                            }
